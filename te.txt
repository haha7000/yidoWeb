OcrMasterProject/
├── app/
│   ├── main.py               # FastAPI 앱 생성 및 전체 라우터 포함
│   │
│   ├── core/                   # 프로젝트의 핵심 설정
│   │   ├── config.py           # 환경변수 등 설정 관리
│   │   ├── database.py         # 데이터베이스 세션 및 연결 관리
│   │   └── dependencies.py     # 공통 의존성 주입 (e.g., get_current_user)
│   │
│   ├── models/                 # SQLAlchemy DB 모델 (데이터베이스 테이블 정의)
│   │   ├── user_model.py       # User 모델 (공통 사용)
│   │   └── ocr_model.py        # Receipt, Passport, MatchingHistory 등
│   │
│   ├── schemas/                # Pydantic 스키마 (API 데이터 유효성 검사 및 형태 정의)
│   │   ├── token_schema.py     # JWT 토큰 스키마
│   │   ├── user_schema.py      # 사용자 관련 스키마 (생성, 조회 등)
│   │   └── ocr_schema.py       # OCR 요청/응답 스키마
│   │
│   ├── routers/                # API 엔드포인트 (Controllers)
│   │   ├── auth_router.py      # /login, /register, /me 등 인증 API
│   │   └── ocr_router.py       # /ocr/upload, /ocr/results 등 OCR/매칭 관련 API
│   │
│   ├── services/               # 비즈니스 로직
│   │   ├── auth_service.py     # JWT 생성, 사용자 인증 로직
│   │   ├── ocr_service.py      # 이미지 처리, 외부 OCR API 호출 로직
│   │   ├── matching_service.py # OCR 결과와 엑셀 데이터 매칭 로직
│   │   └── archive_service.py  # 작업 내용 아카이빙 로직
│   │
│   ├── repositories/           # 데이터베이스 접근 계층 (DAL)
│   │   ├── user_repository.py  # 사용자 데이터 CRUD
│   │   └── ocr_repository.py   # 영수증, 여권 정보 데이터 CRUD
│   │
│   └── utils/                  # 공통 헬퍼 함수
│       ├── file_utils.py       # ZIP 파일 처리 등 파일 관련 유틸리티
│       └── excel_parser.py     # 매칭용 엑셀 파싱 공통 로직
│
├── tests/                      # 단위/통합 테스트 코드
│   └── test_ocr_api.py
│
├── alembic/                    # 데이터베이스 마이그레이션 (Alembic)
├── uploads/                    # 사용자가 업로드한 파일 임시 저장
├── .env                        # 환경변수 파일 (DB 접속정보, 시크릿 키 등)
├── requirements.txt            # 프로젝트 의존성 패키지 목록
└── README.md                   # 프로젝트 설명
